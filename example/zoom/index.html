Zooming Feature Example

How to Zoom

var wavesurfer = Object.create(WaveSurfer);

wavesurfer.init({
    container: document.querySelector('#wave'),
    backend: 'MediaElement'
});

document.querySelector('#slider').oninput = function () {
    wavesurfer.zoom(Number(this.value));
};
